
[!] Installing:

[1] pkg update -y && pkg upgrade -y
[2] git clone https://github.com/void900/ghost-userbot
[3] cd ghost-userbot
[4] nano config.py


[settings config]

chat_id = chat id
delay = time in seconds


[settings API]

[!] API you can get from my.telegram.org

api_id = api id
api_hash = api hash


[5] pip3 install -r requirements.txt
[6] python3.10 main.py
